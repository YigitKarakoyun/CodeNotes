C++ While Loop Example
Let's see a simple example of while loop to print table of 1.

#include <iostream>  
using namespace std;  
int main() {         
 int i=1;      
         while(i<=10)   
       {      
            cout<<i <<"\n";    
            i++;  
          }       
    }
