using System.Collections.Generic;
using System.Data.Entity;
namespace BA.Entity
{
    public class Ders
    {
        public int DersID { get; set; }
        public string Adi { get; set; }

        public virtual ICollection<Ogretmen> Ogretmens { get; set; }
        public Ders()
        {
            Ogretmens = new HashSet<Ogretmen>();
        }

    }
    public class Ogretmen
    {
        public int OgretmenID { get; set; }
        public string Adi { get; set; }
        public virtual Ders Ders { get; set; }
    }
    public class BAEntities : DbContext
    {
        public BAEntities() : base("name=BAConnection")
        {

        }

        public DbSet<Ders> Ders { get; set; }
        public DbSet<Ogretmen> Ogretmens { get; set; }
    }
}
