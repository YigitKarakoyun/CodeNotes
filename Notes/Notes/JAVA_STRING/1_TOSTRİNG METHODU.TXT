// %s dendiði için toStrig otomatik olarak formatý algýlar ve ekrana yazar.
public class zaman {
    private int ay;
    private int gun;
    private int yil;
    public zaman (int a, int g, int y){
	ay = a;
	gun = g;
	yil = y;
	System.out.printf("This için constractor %s\n",this);
    }
    public String toString(){
	return String.format("%d/%d/%d", gun,ay,yil);
    }
}
