public int[] selectionSort(int[] array){
    for (int i=0; i<array.length-1; i++) {
        for (int j=i+1; j<array.length; j++) {
            if (array[i] > array[j]) {
              //elemanlarýn yeri deðiþtiriliyor.
                int temp = array[i];
                array[i] = array[j];
                array[j] = temp;
            }
        }
    }
    return null;
}
